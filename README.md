#MERGE CONFLICT
>git checkout -b quick
>git status
>git diff
>git commit -am "modified"(only for modified files ,not for newly created files.)
>git checkout master
>git checkout quick
>git commit -am "added there"
>git checkout quick
>git diff master
>git merge master
>git status
>git diff
>git commit -am "updated with master"
